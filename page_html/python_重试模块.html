import requests

from retrying import retry



count=0

@retry(stop_max_attempt_number=4)#重试次数4

def test():

    global count

    print('run')

    if count!=5:

        count=1

        raise  Exception('出错了') #重试4次后，仍不成功，给予提示,如果在规定的重试次数内重试成功，就不在提示



test() #调取函数
