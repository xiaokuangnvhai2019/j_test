二分算法:

def two_mothod(num,li,start,end):

    mid=int((end+start)/2)

    if start<=end:

        if (num>li[mid]):

            return two_mothod(num, li, mid+1, end)

        if(num<li[mid]):

            return two_mothod(num,li,start,mid-1)

        if(num==li[mid]):

            return mid

    else:

        return False
